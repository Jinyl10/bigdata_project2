create table sigun(
sigun_id int primary key auto_increment
, SIGUN_CD varchar(10)
, SIGUN_NM varchar(10) 
) default character set utf8;

create table category(
category_id  int  primary key auto_increment 
, category_name varchar(20)
) default character set utf8;

create table restrt(
restrt_id int  primary key auto_increment 
, sigun_id int
, category_id int 
, RESTRT_NM varchar(30)
, REFINE_ROADNM_ADDR varchar(50)
, constraint fk_sigun_id foreign key (sigun_id) references sigun(sigun_id)
, constraint fk_category_id foreign key (category_id) references category(category_id)
) default character set utf8;

create table map(
map_id int primary key auto_increment
, restrt_id int 
, REFINE_WGS84_LAT varchar(30)
, REFINE_WGS84_LOGT varchar(30) 
, constraint fk_restrt_id foreign key (restrt_id) references restrt(restrt_id)
) default character set utf8;

create table member(
member_id int primary key auto_increment 
, email varchar(30)
, password varchar(20)
) default character set utf8;

create table score(
score_id int primary key auto_increment
, restrt_id int 
, member_id int
, star int default 0
, constraint fk_restrt_id foreign key (restrt_id) references restrt(restrt_id)
, constraint fk_member_id foreign key (member_id) references member(member_id) 
) default character set utf8;

create table review(
review_id int primary key auto_increment
, restrt_id int 
, member_id int
, review_title(30)
, review_content text
, review_regdate timestamp default now()
, hit int
, filename varchar(30) 
, constraint fk_restrt_id foreign key (restrt_id) references restrt(restrt_id)
, constraint fk_member_id foreign key (member_id) references member(member_id) 
) default character set utf8;

create table review_img(
review_img_id int primary key auto_increment
, review_id int
, img varchar(30)
, constraint fk_review_id foreign key (review_id) references review(review_id)
) default character set utf8;

create table notice(
notice_id int primary key auto_increment
, notice_title varchar(30)
, notice_content text
, notice_regdate timestamp default now();
